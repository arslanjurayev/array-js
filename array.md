1## For görnüşinde massivdaki sanlary 1 sanydan çykarýan
```
let arr = [1, 2, 3, 4]
function array(arr){
    for(let i = 0; i < arr.length; i++){
        console.log(arr[i])
    }
    return "isledi"
}
console.log(array(arr))
```
#### Şul mysalyň while görnüşinde işlenilişi
```
let arr = [8, 5, 1, 4]
function array(arr){
    let k = 0;
    while(k != arr.length){
        console.log(arr[k])
        k++
    }
    return "isledi"
}
console.log(array(arr))
```
